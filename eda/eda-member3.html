# Extract the first two rows
year_row <- as.character(unlist(raw_data[1, ]))
type_row <- as.character(unlist(raw_data[2, ]))


Code 
# Find the state column
state_col <- 1

# Find only the columns labeled "Median income"
median_cols <- which(type_row == "Median income")

# Keep State + median income columns only
Household_median <- raw_data[-c(1, 2), c(state_col, median_cols)]


Code 
# Rename columns
income_years <- year_row[median_cols]
colnames(Household_median) <- c("State", income_years)


Code 
Household_median <- Household_median %>%
  mutate(
    across(-State, ~ as.numeric(gsub(",", "", .)))
  )
